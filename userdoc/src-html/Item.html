002/**
003 * Class Item :
004 * 
005 * Creates an Item with a name, a description and a weight.
006 * Also contains informations on the item as is it edible
007 * or is it magic and in that case its power.
008 * Contains the getters and setters for each parameter.
009 *
010 * @author Denoela Guennoc
011 * @version 2019
012 */
013public class Item
014{
015    // variables d'instance - remplacez l'exemple qui suit par le v√¥tre
016    private String aItemName;
017    private String aItemDescription;
018    private int aItemWeight;
019    private boolean aIsMagic;
020    private boolean aIsEdible;
021    private int aPower;
022
023    /**
024     * Constructor of the Item class.
025     */
026    public Item(final String pItemName, final String pItemDescription, final int pItemWeight, final boolean pEdible)
027    {
028        // initialisation des variables d'instance
029        this.aItemName = pItemName;
030        this.aItemDescription = pItemDescription;
031        this.aItemWeight = pItemWeight;
032        this.aIsMagic = false;
033        this.aIsEdible = pEdible;
034        this.aPower = 1;
035    }
036    
037    /**
038     * @return true if the item is magic.
039     * @return false otherwise.
040     */
041    public boolean getIsMagic ()
042    {
043        return this.aIsMagic;
044    }
045    
046    /**
047     * @return true if the item is edible.
048     * @return false otherwise.
049     */
050    public boolean getIsEdible ()
051    {
052        return this.aIsEdible;
053    }
054    
055    /**
056     * Used to access the power of the item.
057     * The power is the number of times the weight the
058     * player can carry would be multiply if the item
059     * is eaten.
060     * @return int aPower
061     */
062    public int getPower ()
063    {
064        return this.aPower;
065    }
066    
067    /**
068     * Used to set an item as a magic one and to set its power.
069     */
070    public void setIsMagic (final boolean pMagic, final int pPower)
071    {
072        this.aIsMagic = pMagic;
073        this.aPower = pPower;
074    }
075    
076    /**
077     * Accessor : used to access the name of the item.
078     * @return String aItemName
079     */
080    public String getItemName ()
081    {
082        return this.aItemName;
083    }
084    
085    /**
086     * Accessor : used to access the description of the item.
087     * @return String aItemDescription
088     */
089    public String getItemDescription ()
090    {
091        return this.aItemDescription;
092    }
093    
094    /**
095     * Accessor : used to access the weight of the item.
096     * @return int aItemWeight
097     */
098    public int getItemWeight ()
099    {
100        return this.aItemWeight;
101    }
102    
103    /**
104     * Modifier : used to set the Item's name.
105     */
106    public void setItemName (final String pItemName)
107    {
108        this.aItemName = pItemName;
109    }
110    
111    /**
112     * Modifier : used to set the Item's description.
113     */
114    public void setItemDescription (final String pItemDescription)
115    {
116        this.aItemDescription = pItemDescription;
117    }
118    
119    /**
120     * Modifier : used to set the Item's weight.
121     */
122    public void setItemWeight (final int pItemWeight)
123    {
124        this.aItemWeight = pItemWeight;
125    }
126}
