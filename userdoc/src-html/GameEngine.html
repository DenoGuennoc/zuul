316    /** 
317     * Checks if the player didn't already exeeded is time limit.
318     * If so, the game ends. If not, this method attributes to each 
319     * command the user gave the right method.
320     */
321     public void processCommand (final String pCommandLine)
322    { 
323        gui.println(pCommandLine);
324        Command aCommand = aParser.getCommand(pCommandLine);
325        
326        if (this.aPlayer.getTimer() > this.aPlayer.aTimeLimit)
327        {
328            gui.println("Time is over, you loose...");
329            gui.println("Try again you may win next time !");
330            endGame();
331            return;
332        }
333        
334         if (aCommand.isUnknown()) 
335        {
336            gui.println("I don't know what you mean...");
337            return;
338        }
339        
340        String vCommandWord = aCommand.getCommandWord();
341        if (vCommandWord.equals("help"))
342            printHelp();
343        else if (vCommandWord.equals("go"))
344            goRoom(aCommand);
345        else if (vCommandWord.equals("quit")) {
346            if(aCommand.hasSecondWord())
347                gui.println("Quit what?");
348            else
349                endGame();
350        }
351        else if (vCommandWord.equals("look")) 
352            look();
353        else if (vCommandWord.equals("eat"))
354            if(!aCommand.hasSecondWord())
355                gui.println("Eat what?");
356            else    
357                eat(aCommand);
358        else if (vCommandWord.equals("back"))
359            if(aCommand.hasSecondWord())
360                gui.println("No second word after back");
361            else
362                back();
363        else if (vCommandWord.equals("test"))
364            if(! aCommand.hasSecondWord())
365                gui.println("Test what ?");
366            else
367                test(aCommand);
368        else if (vCommandWord.equals("take"))
369            if(! aCommand.hasSecondWord())
370                gui.println("Take what ?");
371            else
372                take(aCommand);
373        else if (vCommandWord.equals("drop"))
374            if(! aCommand.hasSecondWord())
375                gui.println("Drop what ?");
376            else
377                drop(aCommand);
378        else if (vCommandWord.equals("inventory"))
379            if(aCommand.hasSecondWord())
380                gui.println("No second word after inventory");
381            else
382                inventory();
383        
384    }

398    /**
399     * Allows to move from a room to another if an exit is available
400     * in the direction chosen by the user.
401     * Also increment the clock each time the player gets to an other
402     * room except if it is one of the Air or Underground transport centers.
403     */
404    private void goRoom (final Command pDep)
405    {
406        // Cas 1 : pas de direction donnée
407        if (!pDep.hasSecondWord())
408        { gui.println("Go where ?"); return; }
409        
410        // Extraction du SecondWord pour connaître la direction choisie
411        String vDirection = pDep.getSecondWord();
412        Room vNextRoom = null;
413        
414        // Ensemble des cas de déplacements possibles
415        if (this.aPlayer.getCurrentRoom().getExit(vDirection) != null) {
416            vNextRoom = this.aPlayer.getCurrentRoom().getExit(vDirection);
417            this.aPlayer.addPreviousRoom(this.aPlayer.getCurrentRoom());
418            this.aPlayer.setCurrentRoom(vNextRoom);
419            this.printLocationInfo();
420            if (pDep.getSecondWord() != "up" && pDep.getSecondWord() != "down")
421                this.aPlayer.Clock();
422        }
423        else
424            gui.println("There is no door !");
425    }

436    /**
437     * By using the eat command, the user can eat the edibles items he
438     * has in his bag.
439     * If this items are magic his strength is tenfold ans he can now
440     * carry even more items with him.
441     */
442    private void eat(final Command pCom)
443    {
444        if (this.aPlayer.getCarriedItems().ItemExist(pCom.getSecondWord()))
445        {
446            if (this.aPlayer.getCarriedItems().getItem(pCom.getSecondWord()).getIsEdible())
447            {
448                if (! this.aPlayer.getCarriedItems().getItem(pCom.getSecondWord()).getIsMagic())
449                    gui.println("You have eaten now and you are not angry any more");
450                else
451                {
452                    gui.println("You've got plenty of energy now !");
453                    Item vItem = this.aPlayer.getCarriedItems().getItem(pCom.getSecondWord());
454                    int vMaxWeight = this.aPlayer.getMaxWeight() * vItem.getPower();
455                    this.aPlayer.setMaxWeight(vMaxWeight);
456                    int vCarriedWeight = this.aPlayer.getCarriedWeight() - vItem.getItemWeight();
457                    this.aPlayer.setCarriedWeight(vCarriedWeight);
458                    this.aPlayer.getCarriedItems().removeItem(vItem);
459                    gui.println("You can carry " + vItem.getPower() + " times more weight than you could before !");
460                }
461            }
462            else
463                gui.println("You can't eat that...");
464        }
465        else
466            gui.println("You don't have that in your bag...");
467    }

469    /**
470     * Allows the user to go back to the previous room by only
471     * entering "back" unless he trys to get back futher than
472     * the first room of the game.
473     * He can neighther go back to a room if there is no door 
474     * between the room he is in and the previous one.
475     */
476    private void back()
477    {
478        if (this.aPlayer.noPreviousRoom())
479            gui.println ("You can't go back any further");
480        else if (!this.aPlayer.canGoBack())
481        {
482            gui.println ("You can't go back to the place you were before this way sorry...");
483        }
484        else
485        {
486            this.aPlayer.setCurrentRoom(this.aPlayer.popPreviousRoom());
487            this.printLocationInfo();
488        }
489    }

491    /**
492     * Allows the player to take any items in the room with
493     * him if it isn't to heavy for him to carry.
494     */
495    private void take (final Command pCom)
496    {
497        if (this.aPlayer.getCurrentRoom().getRoomItems().ItemExist(pCom.getSecondWord()))
498        {
499            Item vItem = this.aPlayer.getCurrentRoom().getRoomItems().getItem(pCom.getSecondWord());
500            int vWeight = this.aPlayer.getCarriedWeight() + vItem.getItemWeight();
501            if (vWeight <= this.aPlayer.getMaxWeight())
502            {
503                this.aPlayer.setCarriedWeight(vWeight);
504                this.aPlayer.getCarriedItems().addItem(vItem);
505                this.aPlayer.getCurrentRoom().getRoomItems().removeItem(vItem);
506                gui.println("You took the " + vItem.getItemName() + " with you");
507                return;
508            }
509            else
510                gui.println ("This Item is to heavy for you to carry");
511        }
512        else
513                gui.println ("There is no such item in here");
514    }

516    /**
517     * Allows the player to drop any items he has in his bag
518     * in his current room.
519     */
520    private void drop (final Command pCom)
521    {
522        if (this.aPlayer.getCarriedItems().ItemExist(pCom.getSecondWord()))
523        {
524            Item vItem = this.aPlayer.getCarriedItems().getItem(pCom.getSecondWord());
525            int vWeight = this.aPlayer.getCarriedWeight() - vItem.getItemWeight();
526            this.aPlayer.setCarriedWeight(vWeight);
527            this.aPlayer.getCarriedItems().removeItem(vItem);
528            this.aPlayer.getCurrentRoom().getRoomItems().addItem(vItem);
529            gui.println("You drop the " + vItem.getItemName());
530            return;
531        }
532        else
533        gui.println ("You don't have such item with you... what do you mean ?");
534    }

536    /**
537     * Used to test some of the game possibilities.
538     * Basically, this method is used to read .txt files and
539     * executes line by line the commands written in it.
540     */
541    private void test (final Command pCom)
542    {
543        File vFile = new File (pCom.getSecondWord() + ".txt");
544        if (!vFile.exists())
545        {
546            gui.println ("The file you're trying to test doesn't exist");
547            return;
548        }
549        else {
550            Scanner vScanner;
551            try {
552                vScanner = new Scanner(vFile);
553                while (vScanner.hasNext()) {
554                    String vLine = vScanner.nextLine();
555                    if(vLine.replaceAll(" ", "").replaceAll("\t", "").startsWith("#")) continue;
556                    processCommand(vLine);
557                }
558            }
559            catch (final FileNotFoundException pFNFE) {
560            }
561        }
562    }
