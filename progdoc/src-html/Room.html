001import java.util.*; 
002
003/**
004 * Room class :
005 * Contains all the needed accessors and modifiers concerning the rooms
006 * and exits.
007 */
008public class Room
009{
010    private String aDescription;
011    private HashMap<String, Room> exits;
012    private String aImageName;
013    private HashSet<Item> aItemList;
014    
015    /**
016     * Constructor of the Room class.
017     */
018    public Room (final String pDescription, final String pImageName) 
019    {
020        this.aDescription = pDescription;
021        exits = new HashMap<String, Room>();
022        this.aImageName = pImageName;
023        this.aItemList = new HashSet<Item>();
024    }
025    
026    /**
027     * Accessor : is used to reach the current room description.
028     */
029    public String getDescription ()
030    {
031        return this.aDescription;
032    }
033    
034    /**
035     * Modifier : set each room exits depending on the name of the exit
036     * and the other room linked to it.
037     */
038    public void setExit (final String pDirection, final Room pVoisin)
039    {
040        exits.put(pDirection, pVoisin);
041    }
042    
043    /**
044     * Accessor : returns a value for each possible exit. Returns null 
045     * if the exit doesn't exist in the current room.
046     */
047    public Room getExit(String pDirection)
048    {
049        return exits.get(pDirection);
050    }
051    
052    /**
053     * Returns a list of all the possible exits for the current room.
054     */
055    public String getExitString()
056    {
057        String returnString = "Exits : ";
058        Set<String> keys = exits.keySet();
059        for (String exit : keys)
060        {
061            returnString += " " + exit;
062        }
063        return returnString;
064    }
065    
066    /**
067     * Returns a longer description of the current room.
068     * It contains the name of the room, the list of its exits
069     * and the list of the items in the room.
070     */
071    public String getLongDescription ()
072    {
073        return "You are in " + aDescription + ".\n" + getExitString() + ".\n" + getItemDescription();
074    }
075    
076    /**
077     * Returns a String describing the room's image name.
078     */
079    public String getImageName ()
080    {
081        return this.aImageName;
082    }
083    
084    /**
085     * Used to add an item to the current room.
086     */
087    public void addItem (final Item pItem)
088    {
089         this.aItemList.add(pItem);
090    }
091    
092    /**
093     * Returns the informations about the items located in the
094     * current room, returns "Item : " + the list of all the 
095     * name of the disponibles items.
096     * 
097     * If there's no item in the room, returns "No item here."
098     */
099    public String getItemDescription()
100    {
101        String vItemList = "Item : ";
102        if (this.aItemList.isEmpty())
103            return "No item here" + ".\n";
104        else   
105            for(Item vItem : aItemList)
106            {
107                vItemList += vItem.getItemName() + " ";
108            }
109        return vItemList + ".\n";
110    }
111    
112} // Room
